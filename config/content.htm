hw1_2

hw1_2: 加入與 Cango 頁面 Fourbar 程式碼相同功能的啟動按鈕, 讓使用者可以利用按鈕啟動或中斷紅色方塊上下碰撞移動. 完成後請將程式碼存入個人 Gist 區.

程式碼說明:

hw1-2.py

# 從 browser 導入 html
from browser import html
# 從 browser 導入 document 並且對應為 doc
from browser import document as doc
# 導入 browser.timer
from browser import timer

# 建立 game-board canvas

canvas = html.CANVAS(width = 400, height = 100)
canvas.id = "game-board"
brython_div = doc["brython_div"]
brython_div <= canvas
ctx = canvas.getContext("2d")
px = 0
py = 50
width = 20
height = 20
speedx = 2
speedy = 2

# 建立 button

brython_div <= html.BUTTON("啟動", id="power")

def cube():
    global px, py, width, height, speedx, speedy
    ctx.clearRect(px, py, width, height)
    ctx.fillStyle = "red"
    px += speedx
    if px > 200 :
        speedx = 0
        py += speedy
    if py < 0 or (py + height) > canvas.height:
        speedy = -speedy
    ctx.fillRect(px, py, width, height)

cube()

# 將 game 設為 None
game = None

def gametion(ev) :
    global game
    # 初始啟動,game 為 None
    if game is None :
         # 每 0.1 秒執行一次 draw 函式繪圖
        game = timer.set_interval(cube, 10)
        # 初始啟動後, 按鈕文字轉為"暫停"
        doc['power'].text = '暫停'
    elif game == 'hold':
        # 當 anim 為 'hold' 表示曾經暫停後的啟動, 因此持續以 set_interval() 持續旋轉, 且將 power 文字轉為"暫停"
        game = timer.set_interval(cube, 10)
        doc['power'].text = '暫停'
    else:
        # 初始啟動後, 使用者再按 power, 此時 anim 非 None 也不是 'hold', 因此會執行 clear_interval() 暫停
        # 且將 anim 變數設為 'hold', 且 power 文字轉為"繼續"
        timer.clear_interval(game)
        game = 'hold'
        doc['power'].text = '繼續'
        
doc["power"].bind("click", gametion)
